library(leaflet)

#providers
leaflet() %>%
  addProviderTiles(provider = "Stadia") %>%
  setView(lng = 5.2913, lat = 52.1326, zoom = 7.2)

  setView(lng = 5.2913, lat = 52.1326, zoom = 7.2)
